#### tools/claude.py
```python
#!/usr/bin/env python3
"""
Claude CLI –æ–±–µ—Ä—Ç–∫–∞ –¥–ª—è —Ä–∞–±–æ—Ç—ã —á–µ—Ä–µ–∑ API –±–µ–∑ –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–π —Å–µ—Å—Å–∏–π
"""
import os
import sys
import anthropic
from pathlib import Path

def get_api_key():
    api_key = os.getenv('ANTHROPIC_API_KEY')
    if not api_key:
        print("‚ùå –û—à–∏–±–∫–∞: ANTHROPIC_API_KEY –Ω–µ –Ω–∞–π–¥–µ–Ω –≤ .env")
        print("–î–æ–±–∞–≤—å—Ç–µ –µ–≥–æ –≤ .env —Ñ–∞–π–ª")
        sys.exit(1)
    return api_key

def read_files(file_paths):
    """–ß–∏—Ç–∞–µ—Ç —Ñ–∞–π–ª—ã –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏—Ö —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ"""
    contents = []
    for path in file_paths:
        try:
            with open(path, 'r', encoding='utf-8') as f:
                content = f.read()
                contents.append(f"### –§–∞–π–ª: {path}\n```\n{content}\n```\n")
        except Exception as e:
            contents.append(f"### –§–∞–π–ª: {path}\n‚ùå –û—à–∏–±–∫–∞ —á—Ç–µ–Ω–∏—è: {e}\n")
    return "\n".join(contents)

def ask_claude(prompt, files=None, model="claude-3-5-sonnet-20241022", max_tokens=4096):
    """–û—Ç–ø—Ä–∞–≤–ª—è–µ—Ç –∑–∞–ø—Ä–æ—Å –∫ Claude —á–µ—Ä–µ–∑ API"""
    client = anthropic.Anthropic(api_key=get_api_key())
    
    # –°–æ–±–∏—Ä–∞–µ–º –ø–æ–ª–Ω—ã–π –ø—Ä–æ–º–ø—Ç
    full_prompt = prompt
    if files:
        files_content = read_files(files)
        full_prompt = f"{prompt}\n\n{files_content}"
    
    # –û—Ç–ø—Ä–∞–≤–ª—è–µ–º –∑–∞–ø—Ä–æ—Å
    try:
        message = client.messages.create(
            model=model,
            max_tokens=max_tokens,
            messages=[
                {"role": "user", "content": full_prompt}
            ]
        )
        
        # –í—ã–≤–æ–¥–∏–º –æ—Ç–≤–µ—Ç
        response_text = message.content[0].text
        print(response_text)
        
        # –ü–æ–∫–∞–∑—ã–≤–∞–µ–º —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É
        input_tokens = message.usage.input_tokens
        output_tokens = message.usage.output_tokens
        
        # –†–∞—Å—á–µ—Ç —Å—Ç–æ–∏–º–æ—Å—Ç–∏ (Sonnet: $3/1M input, $15/1M output)
        cost_input = (input_tokens / 1_000_000) * 3
        cost_output = (output_tokens / 1_000_000) * 15
        total_cost = cost_input + cost_output
        
        print("\n" + "="*60)
        print(f"üìä –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞:")
        print(f"  –í—Ö–æ–¥—è—â–∏–µ —Ç–æ–∫–µ–Ω—ã: {input_tokens:,}")
        print(f"  –ò—Å—Ö–æ–¥—è—â–∏–µ —Ç–æ–∫–µ–Ω—ã: {output_tokens:,}")
        print(f"  üí∞ –°—Ç–æ–∏–º–æ—Å—Ç—å: ${total_cost:.4f}")
        print("="*60)
        
        return response_text
        
    except anthropic.APIError as e:
        print(f"‚ùå API –û—à–∏–±–∫–∞: {e}")
        sys.exit(1)

def main():
    import argparse
    
    parser = argparse.ArgumentParser(
        description='Claude CLI —á–µ—Ä–µ–∑ Anthropic API'
    )
    parser.add_argument('prompt', help='–í–∞—à –≤–æ–ø—Ä–æ—Å –∏–ª–∏ –∑–∞–¥–∞—á–∞')
    parser.add_argument('-f', '--files', nargs='+', help='–§–∞–π–ª—ã –¥–ª—è –∫–æ–Ω—Ç–µ–∫—Å—Ç–∞')
    parser.add_argument('-m', '--model', default='claude-3-5-sonnet-20241022',
                       help='–ú–æ–¥–µ–ª—å Claude (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é: sonnet-3.5)')
    parser.add_argument('-t', '--max-tokens', type=int, default=4096,
                       help='–ú–∞–∫—Å–∏–º—É–º —Ç–æ–∫–µ–Ω–æ–≤ –≤ –æ—Ç–≤–µ—Ç–µ')
    
    args = parser.parse_args()
    
    ask_claude(args.prompt, args.files, args.model, args.max_tokens)

if __name__ == '__main__':
    main()
```
